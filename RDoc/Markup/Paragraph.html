    Joins the raw paragraph text and converts inline HardBreaks to the hard_break text.

    # File lib/rdoc/markup/paragraph.rb, line 18
def text hard_break = ''
  @parts.map do |part|
    if RDoc::Markup::HardBreak === part then
      hard_break
    else
      part
    end
  end.join
end
