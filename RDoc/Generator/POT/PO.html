def add entry
  existing_entry = @entries[entry.msgid]
  if existing_entry
    entry = existing_entry.merge(entry)
  end
  @entries[entry.msgid] = entry
end

def to_s
  po = ''
  sort_entries.each do |entry|
    po += "\n" unless po.empty?
    po += entry.to_s
  end
  po
end
